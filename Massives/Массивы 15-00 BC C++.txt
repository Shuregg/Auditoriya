#include<iostream>
#include<ctime>

using namespace std;

void main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));

	int mas[6];

	for (int i = 0; i < 6; i++)
	{
		mas[i] = rand() % 20 - 10;
		cout << mas[i] << " ";
	}
	int sum = 0;

	for (int i = 0; i < 6; i++)
	{

		if (mas[i] < 0)
		{
			sum = sum + mas[i];

		}
	}

	cout << "Сумма отрицательных элементов " << sum << endl;
	int chetniye = 0;
	for (int i = 0; i < 6; i++)
	{
		if (mas[i] % 2 == 0)
		{
			chetniye = chetniye + mas[i];
		}



	}
	int min = mas[0];
	int max = mas[0];
	int pos_max = 0;
	int pos_min = 0;
	for (int i = 0; i < 6; i++)
	{
		if (mas[i] < min)
		{
			min = mas[i];
			pos_min = i;
		}
	}
	for (int i = 0; i < 6; i++)
	{
		if (mas[i] > max)
		{ 
			max = mas[i];
				pos_max = i;
		}
	}
	int sum_min_max = 0;
	if (pos_max > pos_min)
	{
		swap(pos_max, pos_min);
	}
	for (int i = pos_max + 1; i < pos_min; i++)
	{
		sum_min_max = mas[i] + mas[pos_min - 1];
		
	}
		
		cout << "Min number = " <<min<<endl;
		cout << "Сумма чётных чисел: " << chetniye << endl;
		cout << "Ответ: " << sum_min_max << endl;
	system("pause");
}
