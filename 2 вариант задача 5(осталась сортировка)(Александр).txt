#include<iostream>
#include<ctime>
using namespace std;

void main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));
	const int n = 10;
	float mas[n];
	
	
	for (int i = 0; i < n; i++)
	{
		mas[i] = (rand() % 100 - 50) / 10.0;
		cout << mas[i] <<" ";
	}

	float min = mas[0];
	float max = mas[0];
	float sum_pol = 0;
	for (int i = 0; i < n; i++)
	{
		if( mas[i]> 0)
		{
			sum_pol = sum_pol + mas[i];
		}

	}
	cout << "Сумма чётных чисел в массиве: " <<sum_pol<< endl;
	cout << endl;
	int min_pos = 0;
	int max_pos = 0;
	for (int i = 0; i < n; i++)
	{
		if (mas[i] < min)
		{
			min = mas[i];
			min_pos = i;
		}
	}
	cout << "Минимальное число " <<min<<" с позицией "<<min_pos<< endl;
	for (int i = 0; i < n; i++)
	{
		if (mas[i] > max)
		{
			max = mas[i];
			max_pos = i;
		}
	}
	cout << "Максимальное число " << max << " с позицией " << max_pos << endl;
	float umn_num = 1;
	for (int i = 0; i < n; i++)
	{
		if ((i < max_pos) && (i > min_pos))
		{
			umn_num = umn_num * mas[i];
		}
		if (i < min_pos && i > max_pos)
		{
			umn_num = umn_num * mas[i];
		}
	}
	cout <<"Умножение: "<< umn_num << endl;

	//SSSSSSSSSSSSSSSSSSSSSSSSSSSSSSSSS
	for (int j = 0; j <= n; j++)
	{
		for (int i = 0; i < n - 1; i++)
		{
			if (mas[i] > mas[i + 1])
			{
				swap(mas[i], mas[i + 1]);
			}

		}
	}
	for (int i = 0; i < n; i++)
	{
		cout << mas[i] << " ";
	}
	

	system("pause");
}
