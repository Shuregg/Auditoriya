#include <iostream>
#include<string>

using namespace std;

void main()
{
	setlocale(LC_ALL, "Rus");

	string text;
	cout << "Введите свой текст на наличие определённого слова в нём" << endl;
	getline(cin, text);
	string word;
	cout << "Введите своё слово" << endl;
	getline(cin, word);
	int count = 0;
	for (int i = 0; i < text.length(); i++)
	{
		for (int j = 0; j < word.length(); j++)
		{
			if (text[i] == word[j])
			{
                       
				i++;
				count++;
			}
			if(text[i] != word[j])
			{
				
					count = 0;
					j = 0;
			}
		}
		
	}
	if (count == word.length())
	{
		cout << "Данное слово присутствует в тексте!" << endl;
	}
	else
		{
		cout << "Данное слово отсутсвует в тексте." << endl;
		}
	system("pause");
}
