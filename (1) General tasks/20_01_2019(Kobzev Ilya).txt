float max_mas(float* m, int size)
{
	float max = m[0];
	for (int i = 0; i < size; i++)
	{
		if (m[i] > max)
		{
			max = m[i];
		}
	}
	cout << max << endl;
	return max;
}
float min_mas(float* m, int size)
{
	float min = m[0];
	for (int i = 0; i < size; i++)
	{
		if (m[i] < min)
		{
			min = m[i];
		}
	}
	cout << min << endl;
	return min;
}
float swaping_mas(float* m, int size)
{
	float p;
	float min = m[0];
	for (int i = 0; i < size; i++)
	{
		if (m[i] < min)
		{
			swap(m[i], min);
		}
	}

	cout << m[size] << endl;
	return p, min;
}
